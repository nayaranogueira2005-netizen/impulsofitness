    <script>
        function toggleAdmin() {
            const drawer = document.getElementById('adminDrawer');
            drawer.style.display = (drawer.style.display === 'block') ? 'none' : 'block';
        }

        function updateImages() {
            const ids = ['Hero', 'Staff', 'StaffNutri', 'Staff3', 'Staff4', 'Nutricion'];
            ids.forEach(id => {
                const val = document.getElementById('input' + id).value.trim();
                if (val) {
                    document.getElementById('img' + id).src = val;
                    localStorage.setItem('impulso_v2_' + id.toLowerCase(), val);
                }
            });
            toggleAdmin();
        }

        function resetDefaults() {
            localStorage.clear();
            location.reload();
        }

        window.onload = function() {
            const ids = ['Hero', 'Staff', 'StaffNutri', 'Staff3', 'Staff4', 'Nutricion'];
            ids.forEach(id => {
                const saved = localStorage.getItem('impulso_v2_' + id.toLowerCase());
                if (saved) {
                    document.getElementById('img' + id).src = saved;
                    document.getElementById('input' + id).value = saved;
                }
            });
        };
    </script>
